import torch.nn as nn
import torch.nn.functional as F

class Net1(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(1, 6, 5)           # input channel size, output channel size, kernel size
        self.pool = nn.MaxPool2d(2, 2)            # kernel size : 2x2
        self.conv2 = nn.Conv2d(6, 16, 5)          # input channel size has to be equal to last output channel size, so in our last conv layer, output is even bigger, and kernel size remains the same
        self.fc1 = nn.Linear(16 * 4 * 4, 120)     # first fully connected layer
        self.fc2 = nn.Linear(120, 84)
        self.fc3 = nn.Linear(84, 10)              # output 10 = 10 classes

    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))      # rozmiar: 28-5+1 -> 24 -> /2 ->12
        x = self.pool(F.relu(self.conv2(x)))      # rozmiar 12-5+1 -> 8 -> /2 -> 4 (dlatego w pierwszej linear musimy mieć 4*4*16 channeli)
        x = torch.flatten(x,1)                    # now we need to flatten, -1 - nmbr of betches, potem rozmiar inputu
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.fc3(x)
        return x

class Net2(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(1, 10, 5)
        self.pool = nn.MaxPool2d(2, 2)
        self.conv2 = nn.Conv2d(10, 20, 5)
        self.fc1 = nn.Linear(20 * 4 * 4, 160)
        self.fc2 = nn.Linear(160, 76)
        self.fc3 = nn.Linear(76, 10)

    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))      # 28 -> 12
        x = self.pool(F.relu(self.conv2(x)))      # -> 4
        x = torch.flatten(x,1)
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.fc3(x)
        return x


class Net3(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(1, 6, 3)
        self.pool = nn.MaxPool2d(2, 2)
        self.conv2 = nn.Conv2d(6, 16, 3)
        self.conv3 = nn.Conv2d(16, 32, 3)
        self.avgpool = nn.AdaptiveAvgPool2d((1, 1))
        self.fc1 = nn.Linear(32, 120)
        self.fc2 = nn.Linear(120, 84)
        self.fc3 = nn.Linear(84, 10)

    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))
        #print(x.shape)
        x = self.pool(F.relu(self.conv2(x)))
        #print(x.shape)
        x = self.pool(F.relu(self.conv3(x)))
        #print(x.shape)
        x = torch.flatten(x,1)
        #print(x.shape)
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.fc3(x)
        return x

def train(net, optimizer, epochs=10):
  cross_entrophy = torch.nn.CrossEntropyLoss()
  loss_history = []
  for epoch in range(epochs):
      running_loss = 0
      for i, data in enumerate(trainloader, 0):
          inputs, labels = data
          inputs,labels = inputs.to(x), labels.to(x)
          optimizer.zero_grad()
          outputs = net(inputs)
          loss = cross_entrophy(outputs, labels)
          loss.backward()
          optimizer.step()
          running_loss += loss.item()
          if i % 200 == 199:
            print(f'Epoka {epoch+1}/{epochs}, funkcja strat: {running_loss / 2000:.3f}')
            loss_history.append(running_loss / 2000)
            running_loss = 0.0
  print('Finished Training')
  return loss_history

from torchvision.utils import make_grid
from sklearn.metrics import confusion_matrix, ConfusionMatrixDisplay

def evaluation(net):
    correct = 0
    total = 0
    confusion_matrix = np.zeros((10, 10), dtype=int)
    misclassified_images = []
    with torch.no_grad():
        for data in testloader:
            images, labels = data
            images,labels = images.to(x), labels.to(x)
            outputs = net(images)
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
            for i in range(len(labels)):
                confusion_matrix[labels[i]][predicted[i]] += 1
                if predicted[i] != labels[i] and len(misclassified_images) < 1:     # dodamy jakis jeden obrazek
                    misclassified_images.append((images[i], labels[i], predicted[i]))

    #print(confusion_matrix)
    print(f'Network Accuracy on test images: {100 * correct / total}%')
    plt.figure(figsize=(12, 8))
    ConfusionMatrixDisplay(confusion_matrix,display_labels=labels_map.values()).plot()
    plt.xticks(rotation=45)
    plt.show()

    return misclassified_images[0]

Wnioski¶

Pierwszy model¶

Zwraca bardzo dobre wyniki.

Drugi model¶

Wykazał się nieznacznie lepszą wydajnością w porównaniu do modelu 1. Zastosowanie większej liczby neuronów w warstwach liniowych i filtrów zwracających większą wartość kanałów przyczyniło się do delikatnej poprawy wyników. Funkcja strat początkowo osiągała niższe (lepsze) wartości, ale po 10 epokach, wartość przez nią zwracana jest porównywalna dla obu sieci.

Trzeci model¶

Po dodaniu kolejnej warstwy konwolucyjnej, trzeci model osiągnął najgorsze wyniki. Funkcja strat zwraca wartości dwukrotnie większe niż w przypadku modelu 2. Dokładność dopasowania etykiet na zbiorze testowym była niższa o 5 punktów procentowych w porównaniu do modelu 2 i o 3 punkty procentowe w porównaniu do modelu 1.

Ogólne spostrzeżenia¶

Wszystkie modele miały problem z klasyfikacją podobnych elementów garderoby, tj. z rozróżnianiem górnych elementów garderoby, jak koszule, t-shirty i kurtki. Na przykładowych, źle odgadniętych obrazkach, widzimy, że nawet gołym okiem jest ciężko rozróżnić np. płaszcz od bluzy (przykładowy obrazek dla trzeciego modelu).

from time import time
def train_and_time(net, optimizer, device, epochs=10):
  device = torch.device(device)
  net.to(device)
  start_time = time()
  cross_entrophy = torch.nn.CrossEntropyLoss()
  for epoch in range(epochs):
    for i, data in enumerate(trainloader, 0):
      inputs, labels = data
      inputs, labels = inputs.to(device), labels.to(device)
      optimizer.zero_grad()
      outputs = net(inputs)
      loss = cross_entrophy(outputs, labels)
      loss.backward()
      optimizer.step()
      if i % 500 == 499:
        print(f'Epoka {epoch+1}/{epochs}')
  print('Finished Training')
  end_time = time()
  print(f'Czas uczenia na {device}: {end_time - start_time:.2f} sekund')
